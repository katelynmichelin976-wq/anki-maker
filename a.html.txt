function getQuality(){ return parseInt(document.getElementById('quality-slider').value)/100; }
function updateQualityLabel(v){
  document.getElementById('quality-label').textContent = v+'%';
  const kb = Math.round(4000 * (v/90) * 0.06);
  document.getElementById('est-size').textContent = (kb-20)+'~'+(kb+40)+'KB';
}

// ── Image compression ──────────────────────────────────────────────────────
function compressImage(file, quality, maxPx){
  maxPx = maxPx || 600; // 600px max edge — plenty for Anki card display
  return new Promise(resolve => {
    const reader = new FileReader();
    reader.onload = e => {
      const img = new Image();
      img.onload = () => {
        const scale = Math.min(1, maxPx / Math.max(img.width, img.height));
        const w = Math.round(img.width * scale);
        const h = Math.round(img.height * scale);
        const canvas = document.createElement('canvas');
        canvas.width = w; canvas.height = h;
        const ctx = canvas.getContext('2d');
        // White background (in case of transparent PNG)
        ctx.fillStyle = '#FFFFFF';
        ctx.fillRect(0, 0, w, h);
        ctx.drawImage(img, 0, 0, w, h);
        canvas.toBlob(blob => {
          const reader2 = new FileReader();
          reader2.onload = ev => resolve({
            dataUrl: ev.target.result,
            blob,
            sizeKB: Math.round(blob.size / 1024),
            origSizeKB: Math.round(file.size / 1024),
            w, h
          });
          reader2.readAsDataURL(blob);
        }, 'image/jpeg', quality);
      };
      img.src = e.target.result;
    };
    reader.readAsDataURL(file);
  });
}

// ── Navigation ─────────────────────────────────────────────────────────────
function goStep(n){
  if(n===1 && cards.length===0){ setSt(0,'请先添加卡片'); return; }
  if(n===1) renderReview();
  document.querySelectorAll('.panel').forEach((p,i)=>p.classList.toggle('active',i===n));

}
function setSt(p,msg){ const e=document.getElementById('st'+p); e.textContent=msg; setTimeout(()=>e.textContent='',3500); }

// ── Card management ────────────────────────────────────────────────────────
function addCard(data){
  const id = Date.now()+Math.random();
  cards.push(data||{id,name:'',sensory:'',use:'',context:'',hook:'',

function bindUploads(container, card){
  container.querySelectorAll('.uz').forEach(zone=>{
    const type=zone.dataset.type;
    const input=zone.querySelector('input[type=file]');

    input.addEventListener('change', async e=>{
      const file=e.target.files[0]; if(!file) return;
      await handleFile(file, card, type);
      renderList();
    });

    zone.addEventListener('dragover',e=>{ e.preventDefault(); zone.classList.add('drag'); });
    zone.addEventListener('dragleave',()=>zone.classList.remove('drag'));
    zone.addEventListener('drop',async e=>{
      e.preventDefault(); zone.classList.remove('drag');
      const file=e.dataTransfer.files[0]; if(!file) return;
      await handleFile(file, card, type);
      renderList();
    });
  });
}

async function handleFile(file, card, type){
  if(type==='image'){
    const q=getQuality();
    const result=await compressImage(file, q, 600);
    card.imageDataUrl=result.dataUrl;
    card.imageBlob=result.blob;
    card.imageSizeKB=result.sizeKB;
    card.origSizeKB=result.origSizeKB;
  } else {
    card.audioFile=file;
    card.audioName=file.name;
  }
}

function toggleOpen(id){ const el=document.getElementById('ci-'+id); if(el) el.classList.toggle('open'); }
function upd(id,field,val){ const c=cards.find(c=>String(c.id)===String(id)); if(c) c[field]=val; }
function esc(s){ return String(s||'').replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;'); }
function trunc(s,n){ return s.length>n?s.slice(0,n)+'…':s; }
function fmtKB(kb){ return kb>=1000?(kb/1024).toFixed(1)+'MB':kb+'KB'; }

// ── Review ─────────────────────────────────────────────────────────────────
function renderReview(){
  const list=document.getElementById('rev-list');
  list.innerHTML='';
  let hasErr=false;
  let totalImgKB=0, totalSndKB=0;

  cards.forEach((card,idx)=>{
    const miss=[];
    if(!card.name) miss.push('名称');
    if(!card.imageDataUrl) miss.push('照片');
    if(miss.length) hasErr=true;
    if(card.imageSizeKB) totalImgKB+=card.imageSizeKB;
    if(card.audioFile) totalSndKB+=Math.round(card.audioFile.size/1024);

    const thumb=card.imageDataUrl
      ? `<img src="${card.imageDataUrl}" style="width:60px;height:60px;object-fit:cover;border-radius:8px;border:1px solid var(--border);flex-shrink:0">`
      : `<div style="width:60px;height:60px;background:var(--warm);border-radius:8px;display:flex;align-items:center;justify-content:center;font-size:24px;flex-shrink:0">?</div>`;

    const sizeInfo=card.imageSizeKB
      ? `<span style="font-size:11px;color:var(--green)">📷 ${fmtKB(card.origSizeKB)} → ${fmtKB(card.imageSizeKB)}${card.origSizeKB>card.imageSizeKB?' ✓压缩':''}</span>`
      : `<span style="font-size:11px;color:var(--muted)">（无照片）</span>`;

    const sndInfo=card.audioName
      ? `<span style="font-size:11px;color:var(--green)">🔊 ${esc(trunc(card.audioName,16))} · ${fmtKB(Math.round(card.audioFile.size/1024))}</span>`
      : `<span style="font-size:11px;color:var(--muted)">（无录音，将使用TTS）</span>`;

    const warn=miss.length?`<div style="color:var(--red);font-size:11px;margin-top:4px">⚠ 缺少：${miss.join('、')}</div>`:'';

    const d=document.createElement('div');
    d.className='rev-item';
    d.innerHTML=`${thumb}
      <div style="flex:1;min-width:0">
        <div style="font-size:17px;font-weight:700;font-family:'Noto Serif SC',serif;margin-bottom:4px">${esc(card.name)||'（未填名称）'}</div>
        <div style="display:flex;flex-wrap:wrap;gap:8px">${sizeInfo}${sndInfo}</div>
        ${card.sensory?`<div style="font-size:12px;color:var(--muted);margin-top:3px">外观：${esc(card.sensory)}</div>`:''}
        ${card.hook?`<div style="font-size:11px;color:var(--muted);font-style:italic">「${esc(card.hook)}」</div>`:''}
        ${warn}
      </div>
      <div style="font-size:20px">${miss.length?'⚠':'✓'}</div>`;
    list.appendChild(d);
  });

  // Size summary
  const totalKB=totalImgKB+totalSndKB;
  document.getElementById('size-bar').innerHTML=`
    <div class="size-tag">图片合计：<span>${fmtKB(totalImgKB)}</span></div>
    <div class="size-tag">录音合计：<span>${fmtKB(totalSndKB)}</span></div>
    <div class="size-tag">预计包大小：<span>${fmtKB(totalKB+50)}</span></div>`;

  const errBox=document.getElementById('err-box');
  if(hasErr){ errBox.textContent='部分卡片缺少必填项（名称/照片），请返回补充。'; errBox.classList.add('show'); document.getElementById('gen-btn').disabled=true; }
  else{ errBox.classList.remove('show'); document.getElementById('gen-btn').disabled=false; }
}

// ── Build .apkg ────────────────────────────────────────────────────────────
async function buildApkg(){
  const btn=document.getElementById('gen-btn');
  btn.disabled=true; btn.textContent='生成中…';
  try{
    if(!window.SQL){
      await loadScript('https://cdnjs.cloudflare.com/ajax/libs/sql.js/1.10.2/sql-wasm.min.js');
      window.SQL=await initSqlJs({locateFile:f=>'https://cdnjs.cloudflare.com/ajax/libs/sql.js/1.10.2/'+f});
    }
    if(!window.JSZip) await loadScript('https://cdnjs.cloudflare.com/ajax/libs/jszip/3.10.1/jszip.min.js');

    apkgBlob=await generateApkg();

    let totalImg=0,totalSnd=0;
    cards.forEach(c=>{ if(c.imageSizeKB)totalImg+=c.imageSizeKB; if(c.audioFile)totalSnd+=Math.round(c.audioFile.size/1024); });
    document.getElementById('result-detail').innerHTML=
      `共 <strong>${cards.length}</strong> 张卡片 &nbsp;|&nbsp; `+
      `图片 <strong>${fmtKB(totalImg)}</strong> &nbsp;|&nbsp; `+
      `录音 <strong>${fmtKB(totalSnd)}</strong><br>`+
      `包体积约 <strong>${fmtKB(Math.round(apkgBlob.size/1024))}</strong>`;
    goStep(2);
  }catch(e){
    const eb=document.getElementById('err-box'); eb.textContent='生成失败：'+e.message; eb.classList.add('show');
  }finally{ btn.disabled=false; btn.textContent='✦ 生成 .apkg 文件'; }
}

      card._sndFname=fname;

      mediaFiles.push({idx:mIdx,fname,buf:await card.audioFile.arrayBuffer()});
      mIdx++;
    } else { card._sndFname=null; }
  }

  const css=`.card{font-family:'PingFang SC','Heiti SC','Microsoft YaHei',sans-serif;text-align:center;background:#FFFFFF!important;color:#222!important;padding:16px}.night_mode .card{background:#FFFFFF!important;color:#222!important}.back-box{background:#FAFAF5!important;border:1px solid #DDDDCC;border-radius:12px;padding:16px 22px;margin:8px auto;display:inline-block;text-align:left;max-width:480px;width:90%}.item-name{font-size:34px;font-weight:bold;color:#1A1A1A!important;display:block;text-align:center;margin-bottom:10px;padding-bottom:8px;border-bottom:1px solid #DDD;letter-spacing:4px}.row{font-size:18px;color:#333!important;line-height:2;display:block;padding:2px 0}.lbl{color:#999!important;font-size:15px;margin-right:4px}.hook{font-size:16px;color:#666!important;display:block;padding:6px 0 2px;border-top:1px dashed #DDD;margin-top:6px;font-style:italic}hr#answer{border:none;border-top:1px solid #EEE;margin:10px 0}img{max-width:260px;max-height:260px;border-radius:12px;object-fit:contain}`;
  const qfmt='<div style="text-align:center;padding:12px">{{Front}}</div>';
  const afmt='<hr id="answer">\n{{tts zh_CN:Name}}\n<div class="back-box"><span class="item-name">{{Name}}</span>{{Back}}</div>';

  const modelObj={[MODEL_ID]:{id:MODEL_ID,name:'认知训练自制v2',type:0,mod:now,usn:-1,sortf:0,did:null,
    tmpls:[{name:'卡片',ord:0,qfmt,afmt,bqfmt:'',bafmt:'',did:null,bfont:'',bsize:0}],

          {name:'Name',ord:1,sticky:false,rtl:false,font:'Arial',size:20,media:[]},

    css,latexPre:'',latexPost:'',tags:[],vers:[]}};

    collapsed:false,browserCollapsed:false,newToday:[0,0],revToday:[0,0],lrnToday:[0,0],timeToday:[0,0],dyn:0,conf:1,mod:now},
    '1':{id:1,name:'Default',conf:1,extendRev:50,usn:0,collapsed:false,browserCollapsed:false,
         newToday:[0,0],revToday:[0,0],lrnToday:[0,0],timeToday:[0,0],dyn:0,mod:now}};

    curModel:String(MODEL_ID),collapseTime:1200};

  const db=new SQL.Database();

  return await zip.generateAsync({type:'blob',compression:'DEFLATE',compressionOptions:{level:6}});
}

function loadScript(src){
  return new Promise((res,rej)=>{
    if(document.querySelector(`script[src="${src}"]`)){res();return;}
    const s=document.createElement('script');
    s.src=src;s.onload=res;s.onerror=rej;document.head.appendChild(s);
  });
}
